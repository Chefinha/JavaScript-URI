var input = require('fs').readFileSync('/dev/stdin', 'utf8');
var lines = input.split('\n');

var n = Number(lines[0]);

var ano = n / 365;
var mes = ( n % 365) / 30;
var dia = ( n % 365 ) % 30;

console.log(+ Math.trunc(ano) + " ano(s)");
console.log(+ Math.trunc(mes) + " mes(es)");
console.log(+ Math.trunc(dia) + " dia(s)");
