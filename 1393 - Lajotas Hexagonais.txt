var input = require('fs').readFileSync('/dev/stdin', 'utf8');
var lines = input.split('\n');

var cont = 0;
var resultado,i ,s;

while(Number(lines[cont]) != 0){
    resultado = fibonacci(Number(lines[cont]));
    console.log(resultado);
    cont++;
}
function fibonacci(number){
    s = [1 , 1];
    
    for(i = 2; i <= number; i++){
        s[i] = s[i - 1] + s[i - 2];
    }
    return s[number];
}
