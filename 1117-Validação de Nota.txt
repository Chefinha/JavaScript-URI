var input = require('fs').readFileSync('/dev/stdin', 'utf8');
var lines = input.split('\n');

i = 0;
var cont = 1;
var soma = 0;

while(i <= cont){
var n = Number(lines[i]);
if(n >= 0 && n <= 10){
soma = soma + n;
}else{
console.log("nota invalida");
cont++;
}
i++;
}
var m = soma/2;
console.log("media = " + m.toFixed(2));
