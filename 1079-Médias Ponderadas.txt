var input = require('fs').readFileSync('/dev/stdin', 'utf8');
var lines = input.split('\n');

var n = Number(lines[0]);
var i = 1;

while(i <= n){
var l1 = lines[i].split(' ');
n1 = Number(l1[0]);
n2 = Number(l1[1]);
n3 = Number(l1[2]);
var media = ((2 * n1) + (3 * n2) + (5 * n3))/10;
console.log(media.toFixed(1));
i++;
}
