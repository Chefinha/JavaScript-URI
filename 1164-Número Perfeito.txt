var input = require('fs').readFileSync('/dev/stdin', 'utf8');
var lines = input.split('\n');

var n = Number(lines[0]); //numeros de casos de teste

for(var i = 1; i <= n; i++){ //repetição para pegar os casos de teste
var numPerfeito = Number(lines[i]);
var som = 0;
for(var j = 1; j < numPerfeito; j++){ //repetição para encontrar os divisores
if(numPerfeito % j === 0){
som = som + j;
}
}
if(som == numPerfeito){
console.log(numPerfeito + " eh perfeito");
}else{
console.log(numPerfeito + " nao eh perfeito");
}
}
